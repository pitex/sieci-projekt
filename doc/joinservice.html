type Server struct {
    //	Server IP.
    Address string

    //	Network parent IP.
    Parent string

    //	Children IPs.
    Children []string

    //	How many children already exist.
    ChildNumber int

    // If the server is root, it is pointer to node representing it, otherwise it is nil.
    Root *tree.Node
}
